import java.util.Stack;

class Solution {
    public String solution(int n) {
        String answer = "";
        Stack<Integer> stack = new Stack<>();

        while(n>0){
            int sub = n%3;
            n=(n-1)/3;
            if(sub==0) 
                sub=4;
            stack.push(sub);
        }

        while(!stack.isEmpty())
            answer+=Integer.toString(stack.pop());
        
        return answer;
    }
}

            if(sub==0){
                stack.push(4);
                n--;
            }else
                stack.push(sub);
        }

            answer+=Integer.toString(stack.pop());
