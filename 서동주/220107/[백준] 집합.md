
> **[[백준] 집합](https://www.acmicpc.net/problem/11723)**

	메모리 초과..
	
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
        static HashSet<Integer> set = new HashSet<>();
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        //StringTokenizer st = new StringTokenizer(br.readLine());
        int M = Integer.parseInt(br.readLine());
        
        for(int i = 0; i<M; i++){
            StringTokenizer st = new StringTokenizer(br.readLine());
            String com = st.nextToken();
            
            if(com.startsWith("ad")){
                set.add(Integer.parseInt(st.nextToken()));
            }else if(com.startsWith("r")){
                int n = Integer.parseInt(st.nextToken());
                if(set.contains(n)) set.remove(n);
            }else if(com.startsWith("c")){
                int n = Integer.parseInt(st.nextToken());
                if(set.contains(n)) bw.write(1+"\n");
                else bw.write(0+"\n");
            }else if(com.startsWith("t")){
                int n = Integer.parseInt(st.nextToken());
                if(set.contains(n)) set.remove(n);
                else set.add(n);
            }else if(com.startsWith("al")){
                for(int j = 1; j<=20; j++) set.add(i);
            }else{
                set.clear();
            }
        }
        bw.flush();
        bw.close();

    }
}
```
