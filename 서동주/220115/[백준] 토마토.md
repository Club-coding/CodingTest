
> **[[백준] 토마토](https://www.acmicpc.net/problem/7569)**


```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[] py;
    static int[] px;
    static int M,N,H;
    static Queue<int[]> q;
    static int[][] tomato;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        M = Integer.parseInt(st.nextToken());
        N = Integer.parseInt(st.nextToken());
        H = Integer.parseInt(st.nextToken());
        py = new int[] {-1, 1, 0, 0, N, -N};
        px = new int[] {0, 0, 1, -1, 0, 0};
        tomato = new int[N*H][M];
        
        q = new LinkedList<>();
        
        int cnt = 0;
        for(int i = 0; i<N*H; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j<M; j++){
                int t= Integer.parseInt(st.nextToken());
                tomato[i][j] = t;
                if(t==1) q.offer(new int[] {i, j}); //익은 토마토 넣기
                if(t==0) cnt++; //다 익어있는지 확인
            }
        }
        
        int days = 0; //다 익었으면 그대로 출력
        
        if(cnt>0){ //다 안익어있음
            bfs();
            days = check();
        }

        bw.write(days+"");
        bw.flush();
        bw.close();
    }
    
    static void bfs(){
    
        while(!q.isEmpty()){
            int[] node = q.poll();
            for(int i = 0; i<4; i++){
                int ny = node[0] + py[i];
                int nx = node[1] + px[i];
                //현재 노드 층의 범위
                if(ny >= node[0]/N*N && nx>=0 && ny < (node[0]/N*N+N) && nx<M){
                    if(tomato[ny][nx]==0){
                        tomato[ny][nx] = tomato[node[0]][node[1]]+1;
                        q.offer(new int[] {ny, nx});
                    }
                }
            }
            
            for(int i = 4; i<6; i++){
                int ny = node[0] + py[i];
                int nx = node[1] + px[i];
                //위 아래 영향 확인
                if(ny>=0 && nx>=0 && ny<N*H && nx<M){
                    if(tomato[ny][nx]==0){
                        tomato[ny][nx] = tomato[node[0]][node[1]]+1;
                        q.offer(new int[] {ny, nx});
                    }
                }
            }
        }
        
    }
    //안익은 토마토 확인, 최댓값 확인
    static int check(){
        int max = 0;
        for(int i = 0; i<N*H; i++){
            for(int j = 0; j<M; j++){
                if(tomato[i][j]==0) return -1; ///안익은게 있으면
                if(max<tomato[i][j]) max = tomato[i][j];
            }
        }
        return max-1;
    }
    
}
```
