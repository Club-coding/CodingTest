> **[주사위 굴리기](https://www.acmicpc.net/problem/14499)**

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.util.*;

public class Main{
    public static void main(String[] args) throws IOException{
        int[] dice = new int[7]; //1 : 동, 2 : 서, 3 : 북, 4 : 남
        int[] dx = {0, 1, -1, 0, 0}; //동, 서
        int[] dy = {0, 0, 0, -1, 1}; //북, 남
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int N = Integer.parseInt(st.nextToken()); //세로
        int M = Integer.parseInt(st.nextToken()); //가로
        int y = Integer.parseInt(st.nextToken()); 
        int x = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken()); //이동 횟수
        
        int[][] map = new int[N][M];
        //지도 초기화
        for(int i = 0; i < N; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j < M; j++){
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        
        st = new StringTokenizer(br.readLine());
        
        
        for(int i = 0; i < K; i++){
            int com = Integer.parseInt(st.nextToken()); //이동방향
            //맵 밖 이동시 명령어 실행 X
            if(x+dx[com] >= M || x+dx[com] < 0 || y + dy[com] >= N || y + dy[com] < 0) continue;
            x += dx[com]; y += dy[com]; //지도 이동
            if(com == 1){ //지도의 가로면 동쪽
                int temp = dice[3];
                dice[3] = dice[1];
                dice[1] = dice[4];
                dice[4] = dice[6];
                dice[6] = temp;  
            }else if(com == 2){ //지도의 가로면 서쪽
                int temp = dice[4];
                dice[4] = dice[1];
                dice[1] = dice[3];
                dice[3] = dice[6];
                dice[6] = temp;
            }else if(com == 3){ //지도의 세로면 북쪽
                int temp = dice[2];
                dice[2] = dice[1];
                dice[1] = dice[5];
                dice[5] = dice[6];
                dice[6] = temp;
                
            }else if(com == 4){ //지도의 세로면 남쪽
                int temp = dice[6];
                dice[6] = dice[5];
                dice[5] = dice[1];
                dice[1] = dice[2];
                dice[2] = temp;
            }
            
            if(map[y][x] == 0){
                map[y][x] = dice[6]; //주사위 숫자 복사
            }else{
                dice[6] = map[y][x]; //지도 숫자 복사
                map[y][x] = 0;
            }
            System.out.println(dice[1]);
        }
    }
}
```

> **[주사위 굴리기2](https://www.acmicpc.net/problem/23288)**
	
	시계방향으로 회전하기 위해서 순서대로 dx,dy 배열 배치

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.util.*;

public class Main{
    static int[] dx = {1, 0, -1, 0}; //0 : 동 // 1 : 남 // 2 : 서 // 3: 북
    static int[] dy = {0, 1, 0, -1};
    
    public static void main(String[] args) throws IOException{
        int[] dice = {0, 1, 2, 3, 4, 5, 6};
        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int N = Integer.parseInt(st.nextToken()); //세로
        int M = Integer.parseInt(st.nextToken()); //가로
        int cnt = Integer.parseInt(st.nextToken()); //이동횟수
        int y = 0;
        int x = 0;
        int com = 0; //이동방향
        int sum = 0;
        int[][] map = new int[N][M];
        for(int i = 0; i < N; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j < M; j++){
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        
        for(int i = 0; i < cnt; i++){
            if(x+dx[com] >= M) com = 2; //진행방향이 막혔을시 반대방향으로 설정
            else if(x+dx[com] < 0) com = 0;
            else if(y+dy[com] >= N) com = 3;
            else if(y+dy[com] < 0) com = 1;
            x += dx[com]; y += dy[com];
            
            if(com == 0){ 
                int temp = dice[3]; //동
                dice[3] = dice[1];
                dice[1] = dice[4];
                dice[4] = dice[6];
                dice[6] = temp;   
            }else if(com == 1){
                int temp = dice[6]; //남
                dice[6] = dice[5];
                dice[5] = dice[1];
                dice[1] = dice[2];
                dice[2] = temp;
            }else if(com == 2){
                int temp = dice[4]; //서
                dice[4] = dice[1];
                dice[1] = dice[3];
                dice[3] = dice[6];
                dice[6] = temp;
            }else if(com == 3){
                int temp = dice[2]; //북
                dice[2] = dice[1];
                dice[1] = dice[5];
                dice[5] = dice[6];
                dice[6] = temp;
            }
            
            int B_cnt = bfs(map[y][x], x, y, N, M, map);
            sum += map[y][x] * B_cnt;
			//주사위 진행방향 설정
            if(dice[6] > map[y][x]){ //시계방향
                com++;
                if(com>3) com = 0;
            }else if(dice[6] < map[y][x]){ //반시계방향
                com--;
                if(com<0) com = 3;
            }
        }
        
        System.out.println(sum);
        
    }
    
    public static int bfs(int B, int x, int y, int N, int M, int[][] map){
        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[N][M];
        queue.offer(new int[]{x, y});
        int cnt = 0;
        while(!queue.isEmpty()){
            int[] loc = queue.poll();
            cnt++;
            visited[loc[1]][loc[0]] = true; //초기 B true
            for(int i = 0; i<4; i++){
                int px = loc[0] + dx[i]; 
                int py = loc[1] + dy[i];
                if(px < M && px >= 0 && py < N && py >= 0){
                    if(!visited[py][px] && map[py][px] == B){
                        queue.offer(new int[]{px, py}); //B와같으면 offer
                        visited[py][px] = true;
                    }
                }
            }
        }
        return cnt;
    }
}
```
