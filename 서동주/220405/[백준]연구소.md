

> **[[백준] 연구소](https://www.acmicpc.net/problem/14502)**

```
 * 벽을 세울 모든 경우의 수를 탐색 하기 위해 백트레킹 이용
 * 바이러스 퍼지게 하기 위해 새로운 배열로 복사
 * bfs로 바이러스 퍼지게 실행
 * 완전 탐색으로 안전지역 탐색
```
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[] px = {1, -1, 0, 0};
    static int[] py = {0, 0, 1, -1};
    static int[][] map;
    static int max = 0;
    static int N;
    static int M;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        
        map = new int[N][M];
        
        for(int i = 0; i<N; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j<M; j++){
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        
        dfs(0);
        
        bw.write(max+"");
        bw.flush();
        bw.close();
    }
    
    //벽을 세우기 위한 백트레킹
    public static void dfs(int n){
        if(n==3){
            bfs();
            return;
        }
        
        for(int i = 0; i<N; i++){
            for(int j = 0; j<M; j++){
                if(map[i][j]==0){
                    map[i][j] = 1;
                    dfs(n+1);
                    map[i][j] = 0;
                }
            }
        }
    }
    
    public static void bfs(){
        int[][] copy = new int[N][]; //바이러스를 퍼트리기 위해 복사
        
        for(int i = 0; i<N; i++){
            copy[i] = map[i].clone();
        }
        
        Queue<int[]> queue = new LinkedList<>();
        
        for(int i = 0; i<N; i++){
            for(int j = 0; j<M; j++){
                if(copy[i][j]==2) queue.add(new int[] {i, j}); //바이러스 위치 추가
            }
        }
        
        while(!queue.isEmpty()){
            int[] node = queue.poll();
            
            for(int i = 0; i<4; i++){
                int dy = node[0] + py[i];
                int dx = node[1] + px[i];
                
                if(dy>=0 && dx>=0 && dy<N & dx<M && copy[dy][dx]==0){
                    copy[dy][dx] = 2;
                    queue.add(new int[] {dy, dx});
                }
            }
        }
        
        //안전 구역 탐색
        int cnt = 0;
        for(int i = 0; i<N; i++){
            for(int j = 0; j<M; j++){
                if(copy[i][j]==0) cnt++;
            }
        }
        
        max = Math.max(max, cnt);
    }
}
```
