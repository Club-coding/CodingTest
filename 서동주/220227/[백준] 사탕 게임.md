> **[[백준] 사탕 게임](https://www.acmicpc.net/problem/3085)**

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static char[][] board;
    static int N;
    static int max = 0;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        N = Integer.parseInt(br.readLine());
        board = new char[N][N];
        
        for(int i = 0; i<N; i++){
            board[i] = br.readLine().toCharArray();
        }
        
        //초기 보드 max값 탐색
        ver(0, N, 0, N);
        hor(0, N, 0, N);
        
        change();
        
        bw.write(max+"");
        bw.flush();
        bw.close();
    }
    
    public static void change(){
        for(int i = 0; i<N; i++){
            for(int j = 0; j<N; j++){
                if(i+1<N){
                    if(board[i][j] != board[i+1][j]){ //위아래 교환했을때 위아래 행과 그 열만 비교하면 됨
                        swap(i, j, i+1, j);
                        
                        ver(j, j+1, 0, N);
                        hor(i, i+2, 0, N);
                        
                        swap(i, j, i+1, j);
                    }
                }
                if(j+1<N){
                    if(board[i][j] != board[i][j+1]){ //양옆 교환했을때 양옆 열과 그 행만 비교하면됨
                        swap(i, j, i, j+1);
                        
                        ver(j, j+2, 0, N);
                        hor(i, i+1, 0, N);
                        
                        swap(i, j, i, j+1);
                    }
                }
            }
        }
    }
    
    public static void swap(int i, int j, int y, int x){
        char c = board[i][j];
        board[i][j] = board[y][x];
        board[y][x] = c;
    }
    
    //세로 탐색
    public static void ver(int y1, int y2, int x1, int x2){
        for(int i = y1; i<y2; i++){
            int cnt = 1;
            for(int j = x1; j<x2-1; j++){
                if(board[j][i]==board[j+1][i]) cnt++;
                else cnt = 1;
                max = Math.max(max, cnt);
            }
        }
    }
    
    //가로탐색
    public static void hor(int y1, int y2, int x1, int x2){
        for(int i = y1; i<y2; i++){
            int cnt = 1;
            for(int j = x1; j<x2-1; j++){
                if(board[i][j]==board[i][j+1]) cnt++;
                else cnt = 1;
                max = Math.max(max, cnt);
            }
        }
    }
}
```
