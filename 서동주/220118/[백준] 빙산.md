
> **[[백준] 빙산](https://www.acmicpc.net/problem/2573)**


```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[][] ice;
    static boolean[][] visited;
    static int[] px = {-1, 1, 0, 0};
    static int[] py = {0, 0, -1, 1};
    static int N, M;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        
        ice = new int[N][M];
        
        for(int i = 0; i<N; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j<M; j++){
                ice[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        
        
        int day = 0;
        while(true){
            int cnt = check();
            if(cnt>=2) break; //영역이 두개 이상 나누어지면 정지
            if(cnt==0){ //영역이 0개라면 day = 0 출력
                day = 0;
                break;
            }
            melt();
            day++;
        }

        bw.write(day+"");
        bw.flush();
        bw.close();

    }
    //빙산 녹이기
    public static void melt(){
        Queue<int[]> q = new LinkedList<>();
        int ny = 0;
        int nx = 0;
        for(int i = 0; i<N; i++){
            for(int j = 0; j<M; j++){
                int cnt = 0;
                if(ice[i][j] != 0){ //주위의 0개수 카운트 해서 q에 넣기
                    for(int k = 0; k<4; k++){
                        ny = i + py[k];
                        nx = j + px[k];
                        if(ny>=0 && nx>=0 && ny<N && nx<M){
                            if(ice[ny][nx] == 0) cnt++;
                        }
                    }
                    if(cnt!=0)q.offer(new int[] {i, j, cnt});
                }
            }
        }
        
        //cnt들을 빼주기
        while(!q.isEmpty()){ 
            int[] node = q.poll();
            int minus = ice[node[0]][node[1]] - node[2];
            
            if(minus<0) ice[node[0]][node[1]] = 0;
            else ice[node[0]][node[1]] = minus;
        }
    }
    //빙산 영역 체크
    public static void bfs(int y, int x){
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {y, x});
        
        while(!q.isEmpty()){
            int[] node = q.poll();
            
            for(int i = 0; i<4; i++){
                int ny = node[0] + py[i];
                int nx = node[1] + px[i];
                
                if(ny>=0 && nx>=0 && ny<N && nx<M && !visited[ny][nx] && ice[ny][nx] != 0){
                    q.offer(new int[] {ny, nx});
                    visited[ny][nx] = true;
                }
            }
        }
    }
    //영역 수 체크
    public static int check(){
        int cnt = 0;
        visited = new boolean[N][M];
        for(int i = 0; i<N; i++){
            for(int j = 0; j<M; j++){
                if(ice[i][j] != 0 && !visited[i][j]){
                    visited[i][j] = true;
                    cnt++;
                    bfs(i, j);
                }
            }
        }
        
        return cnt;
    }
}
```
