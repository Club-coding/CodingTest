
> **[[백준] 안전 영역](https://www.acmicpc.net/problem/2468)**


```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int N;
    static int[][] state;
    static boolean[][] visited;
    static int[] px = {-1, 1, 0, 0};
    static int[] py = {0, 0, -1, 1};
    static int max = 0;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st;
        
        N = Integer.parseInt(br.readLine());
        
        state = new int[N][N];
        for(int i = 0; i<N; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 0; j<N; j++){
                state[i][j] = Integer.parseInt(st.nextToken());
                if(max<state[i][j]) max = state[i][j]; //최대 높이
            }
        }
        
        int max_cnt = 0;
        
        for(int i = 0; i<=max; i++){
            visited = new boolean[N][N];
            int cnt = 0;
            for(int j = 0; j<N; j++){
                for(int k = 0; k<N; k++){
                    if(state[j][k]>i && !visited[j][k]){
                        visited[j][k] = true;
                        cnt++; //영역 개수
                        bfs(j, k , i);
                    }
                }
            }
            if(cnt>max_cnt) max_cnt=cnt;
        }
        
        bw.write(max_cnt+"");
        bw.flush();
        bw.close();

    }
    
    public static void bfs(int y, int x, int num){
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {y, x});
        
        while(!q.isEmpty()){
            int[] node = q.poll();
            
            for(int i = 0; i<4; i++){
                int ny = node[0] + py[i];
                int nx = node[1] + px[i];
                
                if(ny>=0 && nx>=0 && ny<N && nx<N && !visited[ny][nx] && state[ny][nx]>num){
                    visited[ny][nx] = true;
                    q.offer(new int[] {ny, nx});
                }
            }
        }
    }
}
```
