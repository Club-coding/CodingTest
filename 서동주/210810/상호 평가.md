> **[상호 평가](https://programmers.co.kr/learn/courses/30/lessons/83201)**

	 * 배열의 최솟값을 left로 잡지 않기
	 * 모든 값을 더해도 예산을 초과하는 경우가 있음
	 * 이분탐색 문제

```java
class Solution {
    public String solution(int[][] scores) {
        String answer = "";
        
        //열로 탐색
        for(int i = 0; i<scores[0].length; i++){
            int self = scores[i][i];
            int sum = 0;
            int max = scores[0][i];
            int min = scores[0][i];
            int len = scores.length;
            
            //열에서 min, max 값을 구함
            for(int  j = 0; j<scores.length; j++){
                if(max<scores[j][i])
                    max = scores[j][i];
                if(min>scores[j][i])
                    min = scores[j][i];
                sum+=scores[j][i];
            }
            
            int cnt = 0;
            //i행 i열인 자기자신과 max가 같고 max와 같은 값이 있으면 카운트
            for(int  j = 0; j<scores.length; j++){
                if((self==max&scores[j][i]==max)|(self==min&scores[j][i]==min))
                    cnt+=1;
            }

			//자기 자신이 유일한 최댓값 또는 최솟값인 경우
            if(cnt==1){
                sum-=self;
                len--;
            }

            answer = answer.concat(grade(sum/len));
        }
        return answer;
    }
    
    //성적 함수
    public String grade(int total){
        String g = "";
        switch (total/10) {
            case 10:
            case 9:
            return "A";
        case 8:
            return "B";
        case 7:
            return "C";
            case 6:
            case 5:
            return "D";
        default:
            return "F";
        }
    }
}
```
