
> **[[백준] 톱니바퀴](https://www.acmicpc.net/problem/14891)**


```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[][] gear;
    static boolean[] visited;
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st;
        
        gear = new int[4][8];
        
        for(int i = 0; i<4; i++){
            String n = br.readLine();
            for(int j = 0; j<8; j++){
                gear[i][j] = n.charAt(j)-'0';
            }
        }
        
        
        int K = Integer.parseInt(br.readLine());
        
        for(int i = 0; i<K; i++){
            st = new StringTokenizer(br.readLine());
            int num = Integer.parseInt(st.nextToken());
            int dir = Integer.parseInt(st.nextToken());
            visited = new boolean[4];
            int l = gear[num-1][6];
            int r = gear[num-1][2];
            
            
            move(num-1, dir);
        }
        
        
        
        int sum = 0;
        if(gear[0][0] == 1) sum+=1;
        if(gear[1][0] == 1) sum+=2;
        if(gear[2][0] == 1) sum+=4;
        if(gear[3][0] == 1) sum+=8;
        
        bw.write(sum+"");
        bw.flush();
        bw.close();
    }
    
    static void move(int num, int dir){
        visited[num] = true;
        if(num+1<4){
            if(!visited[num+1] && gear[num+1][6] != gear[num][2]){
                visited[num+1] = true;
                move(num+1, dir*-1);
            }
        }
        if(num-1>=0){
            if(!visited[num-1] && gear[num-1][2] != gear[num][6]){
                visited[num-1] = true;
                move(num-1, dir*-1);
            }
        }
        
        if(dir==1) right(gear[num]);
        else if(dir==-1) left(gear[num]);
    }
    
    static void right(int[] arr){
        int tmp = arr[7];
        for(int i = 7; i>0; i--){
            arr[i] = arr[i-1];
        }
        arr[0] = tmp;
    }
    
    static void left(int[] arr){
        int tmp = arr[0];
        for(int i = 0; i<7; i++){
            arr[i] = arr[i+1];
        }
        arr[7] = tmp;
    }   
}
```
